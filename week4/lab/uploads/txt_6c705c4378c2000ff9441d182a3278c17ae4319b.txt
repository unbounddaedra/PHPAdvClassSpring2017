package lab03_pt2;

import java.util.Date;
import java.time.LocalDateTime;

public class Account {

	private int id;
	public double balance;
	public double anualIntrestRate = 0.045;
	public Date data;

	public Account() {

	}

	public Account(int id, double initBalance) {
		this.balance = initBalance;
		this.id = id;
		this.data = new java.util.Date();
		
	}

	public double getMonthlyIntrestRate() {
		return anualIntrestRate/12;
	}

	public double getMonthlyIntrest() {
		return balance * getMonthlyIntrestRate();
	}

	public boolean withdrawl(double money) {
		this.balance = this.balance-money;
		return true;
	}

	public boolean deposit(double money) {
		this.balance = this.balance+money;
		return true;
	}

	/**
	 * @return the id
	 */
	public int getId() {
		return id;
	}

	/**
	 * @param id
	 *            the id to set
	 */
	public void setId(int id) {
		this.id = id;
	}

	/**
	 * @return the balance
	 */
	public double getBalance() {
		return balance;
	}

	/**
	 * @param ballance
	 *            the balance to set
	 */
	public void setBallance(double ballance) {
		this.balance = balance;
	}

	/**
	 * @return the anualIntrestRate
	 */
	public double getAnualIntrestRate() {
		return anualIntrestRate;
	}

	/**
	 * @param anualIntrestRate
	 *            the anualIntrestRate to set
	 */
	public void setAnuallIntrestRate(double anualIntrestRate) {
		this.anualIntrestRate = anualIntrestRate;
	}

	/**
	 * @return the data
	 */
	public Date getData() {
		return data;

	}

}
